114 Widget *wg_create(widget_type type,Widget *parent,WgGeometry *geometry,XColor color) {
115 
116  int x,y;
117  unsigned int width,height;
118 
119  wg_resolve_geometry(geometry,parent,&x,&y,&width,&height);
120 
121  // create window
122  Window parent_window;
123  if (parent==NULL)
124  parent_window=DefaultRootWindow(display);
125  else
126  parent_window=parent->w;
127 
128  Window w = XCreateSimpleWindow(display, parent_window,x,y,width,height, 0, NIL, color.pixel);
129 
130  // add override_redirect
131  XSetWindowAttributes attributes = { .override_redirect = True };
132  XChangeWindowAttributes(display,w,CWOverrideRedirect,&attributes);
133 
134  Widget *new_widget = wg_create_from_x(type,w,parent,geometry);
135 
136  // add event
137  long event_mask = 0;
138  switch(type) {
139  case wg_decoration:
140  event_mask = ExposureMask | ButtonPressMask | ButtonReleaseMask |PointerMotionMask | SubstructureRedirectMask | SubstructureNotifyMask;
141  new_widget->on_expose = &draw_widget_decoration;
142  break;
143  case wg_title_bar:
144  event_mask = ExposureMask | ButtonPressMask | ButtonReleaseMask | PointerMotionMask;
145  new_widget->on_expose = &draw_widget_title_bar;
146  break;
147  case wg_button:
148  event_mask = ExposureMask | ButtonPressMask;
149  new_widget->on_expose = &draw_widget_button;
150  break;
151  default:
152  event_mask = ExposureMask | ButtonPressMask ;
153 
154  }
155 
156  XSelectInput(display, new_widget->w , event_mask );
157 
158  XDefineCursor(display, new_widget->w, xcursors[0]);
159 
160  return new_widget;
161 }

170 void wg_destroy(Widget *widget) {
171 
172  printf("wg_destroy %d\n",widget->type);
173 
174  // find childs
175  Window root,parent;
176  Window *children;
177  unsigned int nchildren;
178  XQueryTree(display, widget->w, &root, &parent, &children, &nchildren);
179 
180  for(unsigned int i=0; i<nchildren; i ++) {
181  Widget *child=wg_find_from_window(children[i]);
182  if (!child) continue;
183  wg_destroy(child);
184  }
185 
186 printf("Destroy widget %d\n",widget->type);
187  // first destroy the x window
188  if (widget->type != wg_x11) {
189  XUnmapWindow(display,widget->w);
190  XDestroyWindow(display,widget->w);
191  }
192 
193  // then the widget
194  if (widget->text) free(widget->text);
195  if (widget->wm_window) free (widget->wm_window);
196 
197  tdelete(widget,&widget_list,widget_cmp);
198  free(widget);
199 
200 
201  if (children) XFree(children);
202 
203 
204 }
