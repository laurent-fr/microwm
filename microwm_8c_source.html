105 Status get_window_name(Window w, char **name) {
106  int status;
107  XTextProperty text_prop;
108  char **list;
109  int num;
110 
111  status = XGetWMName(display, w, &text_prop);
112  if (!status || !text_prop.value || !text_prop.nitems) return 0;
113  status = XmbTextPropertyToTextList(display, &text_prop, &list, &num);

366 void on_click_decoration(Widget *decoration,XButtonPressedEvent e) {
367 
368  // do nothing if window is maximized
369  if (decoration->wm_window->state == wm_maximized ) return;
370 
371  XRaiseWindow(display, decoration->w);
372 
373  // get initial mouse_position
374  int x_mouse_init = e.x_root;
375  int y_mouse_init = e.y_root;
376 
377  // get initial decoration window position
378  XWindowAttributes window_init_attrs;
379  XGetWindowAttributes(display,decoration->w,&window_init_attrs);
380  int width_window_init = window_init_attrs.width;
381  int height_window_init = window_init_attrs.height;
382  int x_window_init = window_init_attrs.x;
383  int y_window_init = window_init_attrs.y;
384 
385  // get resize type
386  int position = 0;
387  if (e.x<DECORATION_MARGIN_TOP) position |= west;
388  if (e.x>(width_window_init-DECORATION_MARGIN_TOP)) position |= east;
389  if (e.y<DECORATION_MARGIN_TOP) position |= north;
390  if (e.y>(height_window_init-DECORATION_MARGIN_TOP)) position |= south;
391 
392  // sub event-loop, exits when button mouse is released
393  XEvent event;
394  Bool resizing = True;
395  int x_mouse_current,y_mouse_current,new_x,new_y,new_width,new_height,resize_dir_x,resize_dir_y;
396  while (resizing) {
397  XNextEvent(display, &event);
398 
399  switch (event.type) {
400  case ButtonRelease:
401  resizing = False;
402  break;
403 
404  case MotionNotify:
405  x_mouse_current = event.xmotion.x_root;
406  y_mouse_current = event.xmotion.y_root;
407 
408  new_width = width_window_init;
409  new_height = height_window_init;
410  new_x = x_window_init;
411  new_y = y_window_init;
412  resize_dir_x= 1;
413  resize_dir_y= 1;
414 
415  // calculate the new location and size of the window
416  if (position&north) { new_y+= y_mouse_current - y_mouse_init; resize_dir_y=-1; }
417  if (position&west) { new_x+= x_mouse_current - x_mouse_init; resize_dir_x=-1; }
418  if (position&(north|south)) new_height+=(y_mouse_current - y_mouse_init)*resize_dir_y;
419  if (position&(west|east)) new_width+=(x_mouse_current - x_mouse_init)*resize_dir_x;
420 
421  // minimum values
422  if (new_x<0) new_x=0;
423  if (new_y<0) new_y=0;
424  if (new_width<=2*DECORATION_MARGIN+16) new_width=2*DECORATION_MARGIN+16;
425  if (new_height<=DECORATION_MARGIN+DECORATION_MARGIN_TOP+16) new_height=DECORATION_MARGIN+DECORATION_MARGIN_TOP+16;
426 
427  // move and resize if needed
428  if ((new_width!=width_window_init)||(new_height!=height_window_init)) wg_resize(decoration, new_width, new_height);
429  if ((new_x!=x_window_init)||(new_y!=y_window_init)) wg_move(decoration, new_x, new_y);
430 
431  break;
432 
433  case Expose:
434  on_expose_event(event.xexpose);
435  break;
436  }
437 
438  }

448 void on_click_close(Widget *button,XButtonPressedEvent e) {
449 
450  Window window = button->parent->parent->wm_window->w;
451 
452  // send WM_DELETE_WINDOW to X client
453  XEvent ev;
454  memset(&ev, 0, sizeof (ev));
455 
456  // find supported protocols
457  Atom *protocols,*protocol;
458  int protocols_count=0;
459  XGetWMProtocols(display,window,&protocols,&protocols_count);
460  Bool has_wm_delete_window = False;
461  for (int i=0,protocol=*protocols;i<protocols_count;i ++,protocol++) {
462  //printf("Atom: %s\n",XGetAtomName(display,protocol));
463  char *name = XGetAtomName(display,protocol);
464  int result = strcmp(name,"WM_DELETE_WINDOW");
465  XFree(name);
466  if (!result) { has_wm_delete_window = True ; break ;}
467  }
468  XFree(protocols);
469 
470  if (has_wm_delete_window == True ) {
471  ev.xclient.type = ClientMessage;
472  ev.xclient.window = window;
473  ev.xclient.message_type = XInternAtom(display, "WM_PROTOCOLS", True);
474  ev.xclient.format = 32;
475  ev.xclient.data.l[0] = XInternAtom(display, "WM_DELETE_WINDOW", False);
476  XSendEvent(display, window, False, NoEventMask, &ev);
477  } else {
478  XKillClient(display,window);
479  }
480 
481 }

657 void main_event_loop() {
658 
659  XEvent event;
660 
661  // select events
662  Window root = RootWindow(display, screen_num);
663  XSelectInput(display, root , SubstructureRedirectMask | SubstructureNotifyMask );
664 
665  while (1) {
666  /* Wait for an event */
667  XNextEvent(display, &event);
668 
669  switch (event.type) {
670  case MapNotify:
671  // if (event.xmap.override_redirect == False) {
672  printf("MapNotify %d\n",event.xmap.override_redirect);
673  // create_window_decoration(event.xmap.window);
674  // }
675  break;
676 
677  case MapRequest:
678  printf("Map Request\n");
679  create_window_decoration(event.xmaprequest.window);
680  break;
681 
682  case ConfigureRequest:
683  on_configure_request(event.xconfigurerequest);
684  break;
685 
686  case Expose:
687  on_expose_event(event.xexpose);
688  break;
689 
690  case UnmapNotify:
691  printf("Unmap event\n");
692  on_unmap_event(event.xunmap);
693  break;
694 
695  case DestroyNotify:
696  printf("Destroy event\n");
697  break;
698 
699  case MotionNotify:
700  // discard other motion events
701  while (XCheckTypedEvent(display, MotionNotify, &event));
702  on_motion_event(event.xmotion);
703  break;
704 
705  case ButtonPress:
706  printf("Button Press event\n");
707  on_buttonpress_event(event.xbutton);
708  break;
709 
710  default:
711  printf("Unhandled event %d\n",event.type);
712 
713  }
714  }
